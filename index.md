```python
# Code to check if a string or number is a Palindrome.
class Solution:
    def isPalindrome(self, x: int):

        new_X = str(x)
        if new_X == new_X[::-1]:
            return True
        else:
            return False

check = 1223221
output = Solution()
print(output.isPalindrome(check))
```
